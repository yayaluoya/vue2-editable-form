 */

  /** 控件类型 */

  /** 控件名字 */

  /** 单例对象 */
  static I = new $0();

  formDefaultValue = "";

  renderRaw({ formData }: IConRenderOp) {
    let ref = this.getFormValueRef(formData, this.formDefaultValue);
    return <div></div>;
  }

  getRight(op: IConRightRenderOp) {
    let _ = super.getRight(op);
    _.find((_) => _.title == "常用属性")?.childs!.unshift(...[]);
    return _;
  }
}
